	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = { has_global_flag = uncapped_consecration_initial_config }
	}
	immediate = {
		every_country = {
			limit = { is_ai = no }
			country_event = {
				id = uncapped_consecration.1
			}
			break = yes
		}
		set_global_flag = uncapped_consecration_initial_config
	}
}
